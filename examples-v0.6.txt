-- PART A: INSPECTION

-- ── A1. Object counts by kind ────────────────────────────────

SELECT kind, count(*) AS n
FROM objects
GROUP BY kind
ORDER BY n DESC;


-- ── A2. All basis predicates — arity, arg structure, strictness ─
-- Note: the verification query in common_predicates_kernel.sql
-- uses 'binary'/'ternary' for arity labels (not 'transitive'/
-- 'bitransitive', which describe logical properties, not arity).

SELECT
    o.canonical_name            AS predicate,
    p.arity,
    CASE p.arity
        WHEN 1 THEN 'unary'
        WHEN 2 THEN 'binary'
        WHEN 3 THEN 'ternary'
        ELSE        'higher-order'
    END                         AS arg_structure,
    p.arg_labels,
    p.domain_strictness,
    p.nl_description
FROM predicates p
JOIN objects o ON o.id = p.id
WHERE p.is_basis = true
ORDER BY p.arity, o.canonical_name;


-- ── A3. Full type hierarchy with belief scores ───────────────

SELECT
    child.canonical_name              AS subtype,
    parent.canonical_name             AS supertype,
    round(sb.belief_mean::numeric, 4) AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb  ON sb.id     = s.id
JOIN objects child        ON child.id  = s.object_args[1]
JOIN objects parent       ON parent.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('subtype_of', 'predicate')
ORDER BY parent.canonical_name, child.canonical_name;


-- ── A4. All disjointness axioms ──────────────────────────────
-- v0.6: expanded lattice (27 pairs vs 10 in v0.5).
-- [Fix #12] abstract ⊥ concrete and their immediate children
-- are now seeded explicitly so domain enforcement works without
-- a running reasoner.

SELECT
    a.canonical_name              AS type_a,
    b.canonical_name              AS type_b,
    round(sb.belief_mean::numeric, 4) AS belief
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects a           ON a.id   = s.object_args[1]
JOIN objects b           ON b.id   = s.object_args[2]
WHERE s.predicate_id = stable_uuid('disjoint_with', 'predicate')
ORDER BY type_a, type_b;
-- Expect: 27 rows (v0.5 had 10).


-- ── A5. Source credibility priors ────────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength,
    scs.is_protected
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC;


-- ── A6. type_membership cache — confirm trigger-derived rows ──
-- [Fix #1] type_membership is populated by trg_sync_type_membership,
-- not by direct inserts. This query verifies the cache is live.
-- After common_objects_kernel.sql is applied, expect 25+ rows.

SELECT
    obj.canonical_name  AS object,
    typ.canonical_name  AS type,
    round((tm.alpha / (tm.alpha + tm.beta))::numeric, 4) AS belief
FROM type_membership tm
JOIN objects obj ON obj.id = tm.object_id
JOIN objects typ ON typ.id = tm.type_id
ORDER BY obj.canonical_name, typ.canonical_name;


-- ── A7. Role subtypes (Fix #13) ───────────────────────────────
-- mathematician, programmer, inventor etc. are now seeded as
-- subtype_of(X, role) in common_objects_kernel.sql.

SELECT
    child.canonical_name              AS role_subtype,
    parent.canonical_name             AS parent,
    round(sb.belief_mean::numeric, 4) AS belief
FROM statements s
JOIN statement_belief sb  ON sb.id     = s.id
JOIN objects child        ON child.id  = s.object_args[1]
JOIN objects parent       ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND parent.canonical_name = 'role'
ORDER BY child.canonical_name;
-- Expect: artist, engineer, inventor, mathematician, politician,
--         programmer, scientist, teacher — all belief ≈ 1.0.

-- ── B1. Boolean: "Is a mammal an animal?" ────────────────────

SELECT
    child.canonical_name              AS subtype,
    parent.canonical_name             AS supertype,
    round(sb.belief_mean::numeric, 4) AS belief_mean,
    round(sb.ci_low::numeric, 4)      AS ci_low,
    round(sb.ci_high::numeric, 4)     AS ci_high,
    sb.evidence_strength,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name  = 'mammal'
  AND parent.canonical_name = 'animal';
-- Expect: belief_mean ≈ 1.0000, t_kind = 'eternal'


-- ── B2. Binding: "What is person a subtype of?" ──────────────

SELECT
    parent.canonical_name             AS is_subtype_of,
    round(sb.belief_mean::numeric, 4) AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name = 'person'
ORDER BY sb.belief_mean DESC;
-- Expect: mammal, agent (both near-certain, eternal)


-- ── B3. Reverse binding: "What are direct subtypes of entity?" ─

-- ── B4. Typicality: all typical_of statements ranked ─────────

SELECT
    instance.canonical_name           AS instance,
    type_obj.canonical_name           AS typical_of,
    round(sb.belief_mean::numeric, 4) AS typicality,
    round(sb.ci_low::numeric, 4)      AS ci_low,
    round(sb.ci_high::numeric, 4)     AS ci_high
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects instance    ON instance.id = s.object_args[1]
JOIN objects type_obj    ON type_obj.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('typical_of', 'predicate')
ORDER BY sb.belief_mean DESC;
-- Expect: person→agent highest (~0.90); institution→agent lower (~0.60)

-- ── F1. Forward-chain: "Lovelace is a mammal" ────────────────
-- Chain: lovelace is_a person ∧ person subtype_of mammal
--        → lovelace is_a mammal
--
-- [Fix #6] Use compute_derived_belief() instead of a hardcoded
-- belief_alpha. This encodes the principled rule:
--   result_mean = min(parent_means) * 0.9^chain_length
-- The function returns (alpha, beta) with evidence_strength = 2,
-- reflecting that derived statements start with a weak prior.

DO $$
DECLARE
    ada_id     uuid := stable_uuid('ada_lovelace', 'entity');
    mammal_id  uuid := stable_uuid('mammal',       'entity');
    person_id  uuid := stable_uuid('person',       'entity');
    is_a_id    uuid := stable_uuid('is_a',         'predicate');
    subtype_id uuid := stable_uuid('subtype_of',   'predicate');
    reality_id uuid := stable_uuid('reality',      'context');
    src_stmt_1 uuid;
    src_stmt_2 uuid;
    new_stmt   uuid;
    d_alpha    double precision;
    d_beta     double precision;
BEGIN
    SELECT id INTO src_stmt_1
    FROM statements
    WHERE predicate_id = is_a_id
      AND object_args  = ARRAY[ada_id, person_id]
    LIMIT 1;

    SELECT id INTO src_stmt_2
    FROM statements
    WHERE predicate_id = subtype_id
      AND object_args  = ARRAY[person_id, mammal_id]
    LIMIT 1;

    IF src_stmt_1 IS NULL OR src_stmt_2 IS NULL THEN
        RAISE NOTICE 'Source statements not found; skipping derivation.';
        RETURN;
    END IF;

    -- [Fix #6] Compute derived belief from parents.
    -- 'min' mode: result_mean = min(parent_means) * 0.9^1
    -- Both parents are near-certain (~1.0), so result_mean ≈ 0.9.
    SELECT out_alpha, out_beta
    INTO d_alpha, d_beta
    FROM compute_derived_belief(
        ARRAY[src_stmt_1, src_stmt_2],
        1,       -- chain length (1 inference step)
        'min'    -- conservative combination
    );

    RAISE NOTICE 'compute_derived_belief → alpha=%, beta=% (mean≈%)',
        round(d_alpha::numeric,4),
        round(d_beta::numeric,4),
        round((d_alpha / (d_alpha + d_beta))::numeric,4);

    INSERT INTO statements (
        predicate_id, object_args,
        belief_alpha, belief_beta,
        t_kind, context_id,
        derivation_type, derivation_depth, derived_from
    ) VALUES (
        is_a_id,
        ARRAY[ada_id, mammal_id],
        d_alpha, d_beta,
        'eternal', reality_id,
        'forward_chained', 1,
        ARRAY[src_stmt_1, src_stmt_2]
    )
    RETURNING id INTO new_stmt;

    -- [Fix #11] Authoritative provenance: statement_dependencies is required
    -- for forward_chained statements (enforced by trg_enforce_provenance).
    INSERT INTO statement_dependencies (parent_id, child_id, rule_name, weight)
    VALUES
        (src_stmt_1, new_stmt, 'is_a + subtype_of → is_a', 1.0),
        (src_stmt_2, new_stmt, 'is_a + subtype_of → is_a', 1.0);

    RAISE NOTICE 'Derived statement created: % (lovelace is_a mammal, alpha=%)',
        new_stmt, round(d_alpha::numeric,4);
END $$;


-- ── F2. compute_derived_belief — log_odds mode comparison ─────
-- [Fix #6] Demonstrate both combination strategies on the same parents.
-- log_odds is appropriate when parents are genuinely independent evidence.

WITH parent_ids AS (
    SELECT ARRAY[
        (SELECT id FROM statements
         WHERE predicate_id = stable_uuid('is_a','predicate')
           AND object_args  = ARRAY[stable_uuid('ada_lovelace','entity'),
                                    stable_uuid('person','entity')]
         LIMIT 1),
        (SELECT id FROM statements
         WHERE predicate_id = stable_uuid('subtype_of','predicate')
           AND object_args  = ARRAY[stable_uuid('person','entity'),
                                    stable_uuid('mammal','entity')]
         LIMIT 1)
    ] AS ids
)
SELECT
    'min (conservative)' AS strategy,
    round(out_alpha::numeric,4) AS alpha,
    round(out_beta::numeric,4)  AS beta,
    round((out_alpha/(out_alpha+out_beta))::numeric,4) AS mean
FROM parent_ids, compute_derived_belief(ids, 1, 'min')

UNION ALL

SELECT
    'log_odds (independent)' AS strategy,
    round(out_alpha::numeric,4),
    round(out_beta::numeric,4),
    round((out_alpha/(out_alpha+out_beta))::numeric,4)
FROM parent_ids, compute_derived_belief(ids, 1, 'log_odds');
-- Both strategies give similar results when parents are near-certain.
-- Difference becomes meaningful when parents have moderate belief (0.6–0.8).

FROM conflicts cf
JOIN statements sa ON sa.id = cf.statement_a
WHERE cf.conflict_kind = 'type_mismatch'
  AND cf.statement_a   = cf.statement_b
ORDER BY cf.created_at DESC
LIMIT 3;
-- Expect: one row with conflict_kind='type_mismatch', self-referential,
-- resolution_note describing which arg and expected type were mismatched.
-- The statement remains in the KB for belief attenuation by the reasoner.

-- ── H1. Statement counts by t_kind and derivation_type ───────

SELECT
    t_kind,
    derivation_type,
    count(*)                                        AS n_statements,
    round(avg(sb.belief_mean)::numeric, 4)          AS avg_belief,
    round(avg(sb.evidence_strength)::numeric, 2)    AS avg_evidence
FROM statements s
JOIN statement_belief sb ON sb.id = s.id
GROUP BY t_kind, derivation_type
ORDER BY t_kind, derivation_type;


-- ── H2. [Fix #14] tell_about(person) ─────────────────────────
-- "What does the KB know about person?"
-- tell_about() returns all statements where the entity appears in
-- any arg position, resolved to canonical names, ordered by belief.

SELECT
    predicate,
    arg_names,
    round(belief_mean::numeric, 4)      AS belief,
    evidence_strength,
    t_kind,
    interpretation,
    derivation_type
FROM tell_about(
    stable_uuid('person', 'entity'),
    stable_uuid('reality', 'context'),
    0.0    -- p_threshold: show everything, including kernel axioms
)
ORDER BY belief_mean DESC, predicate;

    t_end
FROM tell_about(

    stable_uuid('reality', 'context'),

-- ── H4. [Fix #14] tell_about with context filter ─────────────
-- Restrict to a specific domain context.

SELECT
    predicate,
    arg_names,
    round(belief_mean::numeric, 4)  AS belief,
    t_kind
FROM tell_about(
    stable_uuid('difference_engine', 'entity'),
    stable_uuid('domain_technology', 'context'),
    0.5

ORDER BY belief_mean DESC;


-- ── H5. Source credibility overview ──────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength,
    scs.is_protected
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC, o.canonical_name;


-- ── H6. Open conflicts ────────────────────────────────────────

SELECT
    cf.id,
    cf.conflict_kind,
    pa.canonical_name                         AS predicate_a,
    round(sba.belief_mean::numeric, 4)        AS belief_a,
    round(sbb.belief_mean::numeric, 4)        AS belief_b,
    cf.resolved,
    cf.resolution_note,
    cf.created_at
FROM conflicts cf
JOIN statements    sa   ON sa.id   = cf.statement_a
JOIN statements    sb_s ON sb_s.id = cf.statement_b
JOIN objects       pa   ON pa.id   = sa.predicate_id
JOIN statement_belief sba ON sba.id = cf.statement_a
JOIN statement_belief sbb ON sbb.id = cf.statement_b
WHERE cf.resolved = false
ORDER BY cf.created_at;


-- ── H7. Equivalent objects ────────────────────────────────────

SELECT
    a.canonical_name                               AS object_a,
    b.canonical_name                               AS object_b,
    round((oe.alpha / (oe.alpha + oe.beta))::numeric, 4) AS equiv_belief,
    oe.alpha + oe.beta                             AS evidence_strength
FROM object_equivalence oe
JOIN objects a ON a.id = oe.object_a
JOIN objects b ON b.id = oe.object_b
ORDER BY equiv_belief DESC;
