-- ── A1. Object counts by kind ────────────────────────────────

SELECT kind, count(*) AS n
FROM objects
GROUP BY kind
ORDER BY n DESC;


-- ── A2. All basis predicates — arity, strictness, inverse ────
-- Includes inverse column (new in v0.7 predicates kernel).

SELECT
    o.canonical_name            AS predicate,
    p.arity,
    CASE p.arity
        WHEN 1 THEN 'unary'
        WHEN 2 THEN 'binary'
        WHEN 3 THEN 'ternary'
        ELSE        'higher-order'
    END                         AS arg_structure,
    p.arg_labels,
    p.domain_strictness,
    CASE WHEN p.inverse_predicate_id IS NOT NULL
         THEN inv.canonical_name
         ELSE NULL
    END                         AS inverse,
    p.nl_description
FROM predicates p
JOIN objects o ON o.id = p.id
LEFT JOIN objects inv ON inv.id = p.inverse_predicate_id
WHERE p.is_basis = true
ORDER BY p.arity, o.canonical_name;
-- Expect: before/after, part_of/has_part, contains/part_of show inverse names.


-- ── A3. Full type hierarchy with effective belief ────────────
-- Uses statement_effective_belief (v0.7) for effective_mean.

SELECT
    child.canonical_name                       AS subtype,
    parent.canonical_name                      AS supertype,
    round(seb.effective_mean::numeric, 4)      AS effective_belief,
    round(seb.raw_mean::numeric, 4)            AS raw_belief,
    s.t_kind,
    s.statement_kind
FROM statements s
JOIN statement_effective_belief seb ON seb.id     = s.id
JOIN objects child                  ON child.id   = s.object_args[1]
JOIN objects parent                 ON parent.id  = s.object_args[2]
WHERE s.predicate_id = stable_uuid('subtype_of', 'predicate')
ORDER BY parent.canonical_name, child.canonical_name;


-- ── A4. All disjointness axioms with severity potential ───────
-- abstract ⊥ concrete and their immediate children.

SELECT
    a.canonical_name                           AS type_a,
    b.canonical_name                           AS type_b,
    round(seb.effective_mean::numeric, 4)      AS effective_belief,
    s.statement_kind
FROM statements s
JOIN statement_effective_belief seb ON seb.id = s.id
JOIN objects a                      ON a.id   = s.object_args[1]
JOIN objects b                      ON b.id   = s.object_args[2]
WHERE s.predicate_id = stable_uuid('disjoint_with', 'predicate')
ORDER BY type_a, type_b;
-- Expect: 27 rows; all statement_kind = 'ontological'.


-- ── A5. Source credibility priors ────────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength,
    scs.is_protected
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC;


-- ── A6. type_membership cache — confirm trigger-derived rows ──
-- Populated by trg_sync_type_membership, never by direct inserts.
-- After common_objects_kernel.sql is applied, expect 25+ rows.

SELECT
    obj.canonical_name  AS object,
    typ.canonical_name  AS type,
    round((tm.alpha / (tm.alpha + tm.beta))::numeric, 4) AS belief
FROM type_membership tm
JOIN objects obj ON obj.id = tm.object_id
JOIN objects typ ON typ.id = tm.type_id
ORDER BY obj.canonical_name, typ.canonical_name;


-- ── A7. Role subtypes ─────────────────────────────────────────
-- mathematician, programmer, inventor etc. seeded as subtype_of(X, role)
-- in common_objects_kernel.sql. [Fix #22] is_a rows for these are NOT
-- in the objects kernel — subtype_of alone is sufficient with recursive
-- CTE domain enforcement.

SELECT
    child.canonical_name                       AS role_subtype,
    parent.canonical_name                      AS parent,
    round(seb.effective_mean::numeric, 4)      AS belief,
    s.statement_kind
FROM statements s
JOIN statement_effective_belief seb ON seb.id     = s.id
JOIN objects child                  ON child.id   = s.object_args[1]
JOIN objects parent                 ON parent.id  = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND parent.canonical_name = 'role'
ORDER BY child.canonical_name;
-- Expect: artist, engineer, inventor, mathematician, politician,
--         programmer, scientist, teacher — all belief ≈ 1.0,
--         statement_kind = 'ontological'.


-- ── A8. [Fix #17] Statement counts by statement_kind ─────────
-- New in v0.7: shows distribution across ontological, empirical,
-- statistical, and rule statements.

SELECT
    statement_kind,
    count(*)                                   AS n_statements,
    round(avg(seb.effective_mean)::numeric, 4) AS avg_effective_belief
FROM statements s
JOIN statement_effective_belief seb ON seb.id = s.id
GROUP BY statement_kind
ORDER BY statement_kind;
-- Expect: ontological (majority), statistical (typicality), rule (FOL rules).

-- ── B1. Boolean: "Is a mammal an animal?" ────────────────────

SELECT
    child.canonical_name                       AS subtype,
    parent.canonical_name                      AS supertype,
    round(seb.effective_mean::numeric, 4)      AS effective_belief,
    round(seb.raw_mean::numeric, 4)            AS raw_belief,
    s.t_kind
FROM statements s
JOIN statement_effective_belief seb ON seb.id     = s.id
JOIN objects child                  ON child.id   = s.object_args[1]
JOIN objects parent                 ON parent.id  = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name  = 'mammal'
  AND parent.canonical_name = 'animal';
-- Expect: effective_belief ≈ 1.0000, t_kind = 'eternal'


-- ── B2. Binding: "What is person a subtype of?" ──────────────

SELECT
    parent.canonical_name                      AS is_subtype_of,
    round(seb.effective_mean::numeric, 4)      AS belief,
    s.t_kind
FROM statements s
JOIN statement_effective_belief seb ON seb.id     = s.id
JOIN objects child                  ON child.id   = s.object_args[1]
JOIN objects parent                 ON parent.id  = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name = 'person'
ORDER BY seb.effective_mean DESC;
-- Expect: mammal, agent (both near-certain, eternal)

-- ── B4. Typicality: all typical_of statements ranked ─────────
-- [Fix #17] All typicality statements have statement_kind = 'statistical'.
-- They are excluded from logical inference chains but visible here directly.

-- ── F1. Forward-chain: "Lovelace is a mammal" ────────────────
-- Chain: lovelace is_a person ∧ person subtype_of mammal
--        → lovelace is_a mammal
--
-- Use compute_derived_belief() instead of a hardcoded belief_alpha.
-- 'min' mode: result_mean = min(parent_means) * 0.9^chain_length.
-- Both parents are near-certain (~1.0), so result_mean ≈ 0.9.
-- [Fix #17] Derived is_a statement uses statement_kind = 'empirical'.

DO $$
DECLARE
    ada_id     uuid := stable_uuid('ada_lovelace', 'entity');
    mammal_id  uuid := stable_uuid('mammal',       'entity');
    person_id  uuid := stable_uuid('person',       'entity');
    is_a_id    uuid := stable_uuid('is_a',         'predicate');
    subtype_id uuid := stable_uuid('subtype_of',   'predicate');
    reality_id uuid := stable_uuid('reality',      'context');
    src_stmt_1 uuid;
    src_stmt_2 uuid;
    new_stmt   uuid;
    d_alpha    double precision;
    d_beta     double precision;
BEGIN
    SELECT id INTO src_stmt_1
    FROM statements
    WHERE predicate_id = is_a_id
      AND object_args  = ARRAY[ada_id, person_id]
    LIMIT 1;

    SELECT id INTO src_stmt_2
    FROM statements
    WHERE predicate_id = subtype_id
      AND object_args  = ARRAY[person_id, mammal_id]
    LIMIT 1;

    IF src_stmt_1 IS NULL OR src_stmt_2 IS NULL THEN
        RAISE NOTICE 'Source statements not found; skipping derivation.';
        RETURN;
    END IF;

    -- Compute derived belief from parents.
    -- 'min' mode: result_mean = min(parent_means) * 0.9^1
    SELECT out_alpha, out_beta
    INTO d_alpha, d_beta
    FROM compute_derived_belief(
        ARRAY[src_stmt_1, src_stmt_2],
        1,       -- chain length (1 inference step)
        'min'    -- conservative combination
    );

    RAISE NOTICE 'compute_derived_belief → alpha=%, beta=% (mean≈%)',
        round(d_alpha::numeric,4),
        round(d_beta::numeric,4),
        round((d_alpha / (d_alpha + d_beta))::numeric,4);

    INSERT INTO statements (
        predicate_id, object_args,
        belief_alpha, belief_beta,
        statement_kind, t_kind, context_id,
        derivation_type, derivation_depth, derived_from
    ) VALUES (
        is_a_id,
        ARRAY[ada_id, mammal_id],
        d_alpha, d_beta,
        'empirical', 'eternal', reality_id,
        'forward_chained', 1,
        ARRAY[src_stmt_1, src_stmt_2]
    )
    RETURNING id INTO new_stmt;

    -- Authoritative provenance: statement_dependencies is required
    -- for forward_chained statements (enforced by trg_enforce_provenance).
    INSERT INTO statement_dependencies (parent_id, child_id, rule_name, weight)
    VALUES
        (src_stmt_1, new_stmt, 'is_a + subtype_of → is_a', 1.0),
        (src_stmt_2, new_stmt, 'is_a + subtype_of → is_a', 1.0);

    RAISE NOTICE 'Derived statement created: % (lovelace is_a mammal, alpha=%)',
        new_stmt, round(d_alpha::numeric,4);
END $$;


-- ── F2. compute_derived_belief — log_odds mode comparison ─────
-- [Fix #21] log_odds mode now correctly ACCUMULATES (sums) log-odds
-- rather than averaging them. Two independent near-certain parents
-- produce a combined result ABOVE 0.9, reflecting the independence
-- signal. (In v0.6, averaging gave the same result as a single parent.)
-- The difference is meaningful when parents have moderate belief (0.6–0.8).

WITH parent_ids AS (
    SELECT ARRAY[
        (SELECT id FROM statements
         WHERE predicate_id = stable_uuid('is_a','predicate')
           AND object_args  = ARRAY[stable_uuid('ada_lovelace','entity'),
                                    stable_uuid('person','entity')]
         LIMIT 1),
        (SELECT id FROM statements
         WHERE predicate_id = stable_uuid('subtype_of','predicate')
           AND object_args  = ARRAY[stable_uuid('person','entity'),
                                    stable_uuid('mammal','entity')]
         LIMIT 1)
    ] AS ids
)
SELECT
    'min (conservative)'    AS strategy,
    round(out_alpha::numeric,4) AS alpha,
    round(out_beta::numeric,4)  AS beta,
    round((out_alpha/(out_alpha+out_beta))::numeric,4) AS mean
FROM parent_ids, compute_derived_belief(ids, 1, 'min')

UNION ALL

SELECT
    'log_odds (accumulate)' AS strategy,
    round(out_alpha::numeric,4),
    round(out_beta::numeric,4),
    round((out_alpha/(out_alpha+out_beta))::numeric,4)
FROM parent_ids, compute_derived_belief(ids, 1, 'log_odds');
-- Both strategies give similar results when parents are near-certain (~1.0).
-- With parents at 0.7 each: min → 0.63; log_odds → noticeably higher (independence signal).

BEGIN

    IF high_id IS NOT NULL AND low_id IS NOT NULL AND high_id <> low_id THEN
        INSERT INTO conflicts (statement_a, statement_b, conflict_kind)
        VALUES (high_id, low_id, 'direct_negation')
        ON CONFLICT DO NOTHING;
        RAISE NOTICE 'Manual conflict registered: % (high) vs % (low)',
            high_id, low_id;
    END IF;
END $$;

UPDATE conflicts

-- ── G5. Soft domain violation example ────────────────────────
-- trg_z_soft_domain_check fires AFTER INSERT OR UPDATE when a
-- statement's arg is not a confirmed member of the predicate's
-- expected arg type (resolved transitively via subtype_of in v0.7).
-- Hard violations raise an exception; soft violations insert a
-- type_mismatch conflict (statement_a = statement_b = the statement).
--
-- Example: agent_of(difference_engine, something).
-- agent_of expects an animate entity as arg[1].
-- difference_engine is an artifact (concrete), not animate.
-- domain_strictness of agent_of = 'soft' → type_mismatch conflict.
-- [Fix #22] The check also fires on UPDATE — demonstrated below.

DO $$
DECLARE
    bad_stmt uuid;
BEGIN
    INSERT INTO statements (
        predicate_id, object_args,
        belief_alpha, belief_beta,
        statement_kind, t_kind, context_id, derivation_type
    ) VALUES (
        stable_uuid('agent_of', 'predicate'),
        ARRAY[
            stable_uuid('difference_engine', 'entity'),  -- NOT animate
            stable_uuid('ada_lovelace',      'entity')
        ],
        3.0, 2.0,
        'empirical', 'always',
        stable_uuid('reality', 'context'),
        'source_ingested'
    )
    RETURNING id INTO bad_stmt;

    RAISE NOTICE 'Soft domain violation statement inserted: %', bad_stmt;
    RAISE NOTICE 'A type_mismatch conflict should now exist.';
END $$;

-- Inspect the auto-generated type_mismatch conflict.
-- statement_a = statement_b = the offending statement (self-referential).
SELECT
    cf.id,
    cf.conflict_kind,
    cf.resolution_note,
    cf.resolved,
    sa.predicate_id,
    sa.object_args
FROM conflicts cf
JOIN statements sa ON sa.id = cf.statement_a
WHERE cf.conflict_kind = 'type_mismatch'
  AND cf.statement_a   = cf.statement_b
ORDER BY cf.created_at DESC
LIMIT 3;
-- Expect: one row, conflict_kind='type_mismatch', self-referential.
-- resolution_note describes which arg and expected type were mismatched
-- (checked transitively via subtype_of in v0.7).


-- ── G6. [Fix #20] Effective belief is penalised by open conflicts ──

  AND s.object_args[2] = stable_uuid('mathematician', 'entity')
ORDER BY s.belief_mean DESC LIMIT 1;
-- Expect: effective_mean < raw_mean when open conflicts exist.
-- After G4 resolves the conflict, effective_mean = raw_mean.

-- ── H6. Source credibility overview ──────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,

ORDER BY scs.mean DESC, o.canonical_name;


-- ── H7. Open conflicts with effective belief impact ───────────
-- [Fix #19/20] Shows severity and effective_mean suppression side by side.

SELECT
    cf.id,
    cf.conflict_kind,
    round(cf.severity::numeric, 4)                AS severity,
    pa.canonical_name                              AS predicate_a,
    round(seba.effective_mean::numeric, 4)         AS effective_mean_a,
    round(seba.raw_mean::numeric, 4)               AS raw_mean_a,
    round(sebb.effective_mean::numeric, 4)         AS effective_mean_b,
    round(sebb.raw_mean::numeric, 4)               AS raw_mean_b,

    cf.created_at
FROM conflicts cf

JOIN statement_effective_belief sebb ON sebb.id = cf.statement_b
WHERE cf.resolved = false
ORDER BY cf.severity DESC NULLS LAST, cf.created_at;
-- Expect: effective_mean_a < raw_mean_a when conflict is open and severe.
